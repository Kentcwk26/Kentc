FUNCTION appendFile(list, listCode)
    l = CALLFUNCTION listIdentifier(listCode)
    OPEN l IN APPEND AS TAppend
        FOR record IN list
            FOR item IN record
                WRITE item TO TAppend
                WRITE "NEWLINE" TO TAppend
            ENDFOR
            WRITE "NEWLINE" TO TAppend
        ENDFOR
    CLOSEFILE
ENDFUNCTION

FUNCTION readFile(listCode):
    l = CALLFUNCTION listIdentifier(listCode)
    OPEN l IN READ AS fRead
        string = READ fRead line-by-line
        FOR record IN string
            stripped = record RIGHTSTRIP("NEWLINE") RIGHTSTRIP(",")
            splitRecord = SPLIT stripped INTO LIST(",")
            PRINT(splitRecord)
        ENDFOR
    CLOSEFILE
ENDFUNCTION

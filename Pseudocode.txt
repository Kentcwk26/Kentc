FUNCTION appendFile(list, listBool):
    IF listBool == TRUE:
        OPEN "tenant.txt" IN APPEND AS Thandler
            for record from list:
                for item in record:
                    WRITE item TO Thandler
                    WRITE "\n" TO Thandler
                WRITE "\n" TO Thandler
    ELSE:
        OPEN "apartment.txt" IN APPEND AS Ahandler 
            for record from list:
                for item in record:
                    WRITE item TO Ahandler
                    WRITE "\n" TO Ahandler
                WRITE "\n" TO Ahandler
        CLOSE apartment.txt
